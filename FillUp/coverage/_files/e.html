434                        }
435                        handPosition += handVelocity * delta;
436                        handVelocity += handAcceleration * delta;
437                        if ((handTarget - handPosition) * direction < 0.01f * direction) {
438                                handPosition = handTarget;
439                                handVelocity = 0.0f;
440                                handAcceleration = 0.0f;
441                                lastHandMoveTime = -1L;
442                        } else {
443                                lastHandMoveTime = System.currentTimeMillis();                                
444                        }
445                        invalidate();
446                } else {
447                        lastHandMoveTime = System.currentTimeMillis();
448                        moveHand();
449                }
450        }

506         * DESCRIPTION:
507         * Called when the screen is touched. Moves the gauge hand accordingly.
508         * @see android.view.View#onTouchEvent(android.view.MotionEvent)
509         */
510        @Override
511        public boolean onTouchEvent(MotionEvent event) {
512                
513                if (!interactive) return false;
514                
515                final String tag = TAG + ".onTouchEvent()";
516                final float EMPTYX = 0.2f;  // the scaled x-coordinate of the E on the gauge background
517                final float FULLX = 0.8f;   // the scaled x-coordinate of the F on the gauge background
518                
519                switch (event.getAction()) {
520            case MotionEvent.ACTION_DOWN:
521            case MotionEvent.ACTION_MOVE:
522            case MotionEvent.ACTION_UP:
523                    
524                        // get the scaled x-coordinate where the screen was touched 
525                    // (0.0 or less = far left, 1.0 or greater = far right)
526                    float x = event.getX() / getWidth();
527                    
528                    // adjust x to keep it within the E-F meter displayed as the background gauge image 
529                    // (if you touch outside the meter, it should snap to the appropriate min/max position)
530                        if (x > FULLX) x = FULLX;
531                        if (x < EMPTYX) x = EMPTYX;
532                        
533                        // calculate the desired gauge hand position from the x value
534                        float position = (x - EMPTYX) / (FULLX - EMPTYX);
535                        //Log.d(tag, String.format("getX()=%f getWidth()=%d x=%f position=%f", event.getX(),getWidth(),x,position));
536                        
537                        // move the hand to the new position
538                        setHandPosition(position);
539                        
540                        // if gesture has finished, notify listener of new hand position 
541                        if ((event.getAction() == MotionEvent.ACTION_UP) && 
542                                (handPositionChangedListener != null)) {
543                                Log.d(tag,String.format("handPosition=%f",handPosition));
544                                handPositionChangedListener.onHandPositionChanged(this,handPosition);
545                        }
546                    return true;
547            }
548            return false;
549        }
