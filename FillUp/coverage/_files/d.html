503     */
504    protected void deleteRow() {
505            
506            // get the record to delete from our list of records
507            GasRecord record = records.get(selectedRow);
508            
509            // attempt to remove the record from the log

515                    
516                    // re-calculate mileage based on the modified data
517                    GasRecordList.calculateMileage(records);
518                    
519                    // update the list view
520                    adapter.notifyDataSetChanged();
521            } else {

523            }
