558            // get a list of records from the log before adding new record
559                List<GasRecord> list = gaslog.readAllRecords(selectedVehicle);
560            
561                // attempt to add the new record to the log
562            if (!gaslog.createRecord(selectedVehicle,record)) { 
563                        Utilities.toast(this,getString(R.string.toast_error_saving_data));
564                    return;
565                }
566            
567            // success!
568            Utilities.toast(this,getString(R.string.toast_data_saved));
569            
570            // need a previous full tank in the log to do any calculations

579            // recalculate mileage
580            GasRecordList.calculateMileage(list);
581            
582            // find the new location of the record in the list (after sort)
583            int location = GasRecordList.find(list,record);
584                        
585            // display mileage calculation if possible
586            if (MileageCalculationDialog.isDisplayable(record)) {
587                    MileageCalculationDialog.init(record);

589                    return;
590            }
