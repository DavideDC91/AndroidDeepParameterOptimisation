790    /*

793     * expensive) if we want to re-use code.  Notice that a call to this method

800    static String formatTotal(boolean decimalFormat, String format, long ttl, long roundMinutes) {
801        if (roundMinutes > 0) {
802            long totalMinutes = ttl / MS_M;
803            ttl = roundMinutes * Math.round((float) totalMinutes / roundMinutes) * MS_M;
804        }
805        long hours = ttl / MS_H;
806        long hours_in_ms = hours * MS_H;
807        long minutes = (ttl - hours_in_ms) / MS_M;
808        long minutes_in_ms = minutes * MS_M;
809        long seconds = (ttl - hours_in_ms - minutes_in_ms) / MS_S;
810        return formatTotal(decimalFormat, format, hours, minutes, seconds, roundMinutes);
811    }
