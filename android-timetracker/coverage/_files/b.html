445     * supplied calendar date falls
446     *
447     * @param tw the day for which to calculate the week start
448     * @param startDay the day on which the week starts. This must be 1-based (1

452    public static Calendar weekStart(Calendar tw, int startDay) {
453        Calendar ws = (Calendar) tw.clone();
454        ws.setFirstDayOfWeek(startDay);
455        // START ANDROID BUG WORKAROUND
456        // Android has a broken Calendar class, so the if-statement wrapping
457        // the following set() is necessary to keep Android from incorrectly
458        // changing the date:
459        int adjustedDay = ws.get(Calendar.DAY_OF_WEEK);
460        ws.add(Calendar.DATE, -((7 - (startDay - adjustedDay)) % 7));
461        // The above code _should_ be:
462        // ws.set(Calendar.DAY_OF_WEEK, startDay);
463        // END ANDROID BUG WORKAROUND
464        ws.set(Calendar.HOUR_OF_DAY, ws.getMinimum(Calendar.HOUR_OF_DAY));
465        ws.set(Calendar.MINUTE, ws.getMinimum(Calendar.MINUTE));
466        ws.set(Calendar.SECOND, ws.getMinimum(Calendar.SECOND));
467        ws.set(Calendar.MILLISECOND, ws.getMinimum(Calendar.MILLISECOND));
468        return ws;

527    private void fillInTasksAndRanges() {
528        // Iterate over each task and set the day values, and accumulate the day 
529        // and week totals
530        Cursor c = db.query(TASK_TABLE, TASK_COLUMNS, null, null, null, null, NAME);
531        // The totals for all tasks for each day, plus one for the week total.
532        long dayTotals[] = {0, 0, 0, 0, 0, 0, 0, 0};
533        if (c.moveToFirst()) {
534            do {
535                int tid = c.getInt(0);
536                String tid_s = String.valueOf(tid);
537                TextView[] arryForDay = dateViews.get(tid);
538                // Fetch an array of times (per day) for the task
539                long[] days = getDays(tid_s);
540                // The total for this task, for the whole week
541                int weekTotal = 0;
542                for (int i = 0; i < 7; i++) {
543                    weekTotal += days[i];
544                    dayTotals[i] += days[i];
545                    
546                    arryForDay[i].setText(days[i] == 0L ? ZERO_TIME : Tasks.formatTotal(decimalTime, FORMAT, days[i], roundMinutes));
547                }
548                // Set the week total.  Since this value can be more than 24 hours,
549                // we have to format it by hand:
550                arryForDay[7].setText(weekTotal == 0L ? ZERO_TIME : Tasks.formatTotal(decimalTime, FORMAT, weekTotal, roundMinutes));
551                dayTotals[7] += weekTotal;
552            } while (c.moveToNext());
553        }
554        c.close();
555 
556        TextView[] totals = dateViews.get(-1);
557        for (int i = 0; i < 7; i++) {
558            totals[i].setText(Tasks.formatTotal(decimalTime, FORMAT, dayTotals[i], roundMinutes));
559        }

561    }

563    /**
564     * Fetch the times for a task within the currently set time range, by day.
565     *
566     * @param tid_s The ID of the task for which to fetch times
567     * @return An array containinging, in each cell, the sum of the times which
568     * fall within that day. Index 0 is the first day starting on the currently
569     * set week. This uses TimeRange.overlap() to make sure that only time that
570     * actually falls on the day is included (even if a particular time entry
571     * spans days).
572     * @see TimeRange.overlap()
573     */
574    private long[] getDays(String tid_s) {
575        Calendar day = Calendar.getInstance();
576        day.setFirstDayOfWeek(startDay);
577        long days[] = {0, 0, 0, 0, 0, 0, 0};
578        Cursor r = db.query(RANGES_TABLE, RANGE_COLUMNS, TASK_ID + " = ? AND "
579                + START + " < ? AND ( " + END + " > ? OR " + END + " ISNULL )",
580                new String[]{tid_s,
581            String.valueOf(weekEnd.getTimeInMillis()),
582            String.valueOf(weekStart.getTimeInMillis())},
583                null, null, null);
584 
585        if (r.moveToFirst()) {
586            do {
587                long start = r.getLong(0);
588                long end;
589                if (r.isNull(1)) {
590                    end = System.currentTimeMillis();
591                } else {
592                    end = r.getLong(1);
593                }
594 
595                day.setTimeInMillis(end);
596 
597                int[] weekDays = new int[7];
598                for (int i = 0; i < 7; i++) {
599                    weekDays[i] = ((weekStart.getFirstDayOfWeek() - 1 + i) % 7) + 1;
600                }
601 
602                // At this point, "day" must be set to the start time
603                for (int i = 0; i < 7; i++) {
604                    Day d = Day.fromCalEnum(weekDays[i]);
605                    day.set(Calendar.DAY_OF_WEEK, d.calEnum);
606                    days[i] += TimeRange.overlap(day, start, end);
607                }
608 
609            } while (r.moveToNext());
610        }
611        r.close();
612        return days;
613    }
