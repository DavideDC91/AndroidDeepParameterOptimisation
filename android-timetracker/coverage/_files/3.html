432        // Inserts an item into the list in order.  Why Java doesn't provide
433        // this is beyond me.
434        private void insert(ArrayList<TimeRange> list, TimeRange item) {
435            int insertPoint = 0;
436            for (; insertPoint < list.size(); insertPoint++) {
437                if (list.get(insertPoint).compareTo(item) != -1) {
438                    break;
439                }
440            }
441            list.add(insertPoint, item);
442            if (insertPoint > 0) {
443                Calendar c = Calendar.getInstance();
444                c.setFirstDayOfWeek(Calendar.MONDAY);
445                TimeRange prev = list.get(insertPoint - 1);
446                c.setTimeInMillis(prev.getStart());
447                int pyear = c.get(Calendar.YEAR),
448                        pday = c.get(Calendar.DAY_OF_YEAR);
449                c.setTimeInMillis(item.getStart());
450                if (pday != c.get(Calendar.DAY_OF_YEAR)
451                        || pyear != c.get(Calendar.YEAR)) {
452                    times.add(insertPoint, new TimeRange(startOfDay(item.getStart()), SEP));
453                }
454            } else {
455                times.add(insertPoint, new TimeRange(startOfDay(item.getStart()), SEP));
456            }
457        }

459        private long startOfDay(long start) {
460            Calendar cal = Calendar.getInstance();
461            cal.setFirstDayOfWeek(Calendar.MONDAY);
462            cal.setTimeInMillis(start);
463            cal.set(Calendar.HOUR_OF_DAY, cal.getMinimum(Calendar.HOUR_OF_DAY));
464            cal.set(Calendar.MINUTE, cal.getMinimum(Calendar.MINUTE));
465            cal.set(Calendar.SECOND, cal.getMinimum(Calendar.SECOND));
466            cal.set(Calendar.MILLISECOND, cal.getMinimum(Calendar.MILLISECOND));
467            return cal.getTimeInMillis();

471            int dayOfYear = -1, year = -1;
472            Calendar curDay = Calendar.getInstance();
473            curDay.setFirstDayOfWeek(Calendar.MONDAY);

480                    dayOfYear = doy;

488        public void updateTimeRange(long sd, long ed, int newTaskId, TimeRange old) {
489            SQLiteDatabase db = dbHelper.getWritableDatabase();
490            ContentValues values = new ContentValues();
491            values.put(START, sd);

493            String whereClause = START + "=? AND " + TASK_ID + "=?";
494            String[] whereValues;
495            if (ed != NULL) {
496                values.put(END, ed);

498                whereValues = new String[]{String.valueOf(old.getStart()),
499                    String.valueOf(currentTaskId),
500                    String.valueOf(old.getEnd())
501                };
502            } else {
503                whereClause += " AND " + END + " ISNULL";
504                whereValues = new String[]{String.valueOf(old.getStart()),
505                    String.valueOf(currentTaskId)
506                };
507            }
508            db.update(RANGES_TABLE, values,
509                    whereClause,
510                    whereValues);
511            if (newTaskId != currentTaskId) {
512                times.remove(old);
513            } else {
514                old.setStart(sd);

516            }
