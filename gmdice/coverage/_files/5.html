404            int val = getSelectedPos(result);
405 
406            /* Ensure the user can't type in a value greater
407             * than the max allowed. We have to allow less than min
408             * as the user might want to delete some numbers
409             * and then type a new number.
410             */
411            if (val > mEnd) {
412                return "";
413            } else {
414                return filtered;
415            }
416        }

420        if (mDisplayedValues == null) {
421            return Integer.parseInt(str);
422        } else {

424

426                str = str.toLowerCase();
427                if (mDisplayedValues[i].toLowerCase().startsWith(str)) {
428                    return mStart + i;
429                }
430            }

432            /* The user might have typed in a number into the month field i.e.

434             */
435            try {
436                return Integer.parseInt(str);
437            } catch (NumberFormatException e) {
438 
439                /* Ignore as if it's not a number we don't care */
440            }
441        }

443    }
